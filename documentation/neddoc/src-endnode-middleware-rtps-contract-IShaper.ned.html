//
// The interface to implement different shaping functionalitites regarding all readers and writers of the middleware module
//

{
    parameters:

        /// It should be possible to enable/disable a shaper
        bool shaperEnabled;
        @display("i=block/join");

    gates:
        /// Input from the participant (dispatcher)
        input dispatcherIn;
        /// Output to the adapter
        output adapterOut;

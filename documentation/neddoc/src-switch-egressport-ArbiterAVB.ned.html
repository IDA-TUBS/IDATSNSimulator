package idaTSNSimulator.switch.egressport;

//
// The ArbiterAVB is a module which is conected to Arbiter module interface 
// The ArbiterAVB arbitrate the in BuffersModule stored Frames and send them out
// Based on the Credit Based Shaper mechanism the frames are selected
//
simple ArbiterAVB like Arbiter
{
    parameters:
        /// the default number of Priorities
        int supportedPriorities;
		/// the default name of AVB configuration file
        string configFileName = default("AVBConfigurationFile.conf");
        @display("i=block/mac;bgb=513,275");

    gates:
        input in;
        output out;
        inout control;
}

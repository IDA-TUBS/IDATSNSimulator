59 /***
60  * Truncates the input with the minimum and maximum.
61  * The maximum is disabled with max=0.
62  */
63 template<typename T> inline T truncate_min_max(T input, T min, T max)
64 {
65  T result = input;
66  if (input < min)
67  {
68  result = min;
69  EV << "-I- " << " Input " << input << " truncated to the minimum " << min << " .\n";
70  }
71  else if (max > 0 && input > max)
72  {
73  result = max;
74  EV << "-I- " << " Input " << input << " truncated to the maximum " << min << " .\n";
75  }
76  return result;
77 }
